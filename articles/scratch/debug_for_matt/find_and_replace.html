#Find and Replace

#Creates output.txt which is a copy of input.txt
#except with 'brown' replace with 'blue'

# open() files for reading and writing
fIn = open('blue.txt', 'r')
fOut = open('brown.txt', 'w')

# fIn and fOut are now Tooltips 'objects', more specifically 'file objects'
# object have attributes, e.g. fIn.mode returns 'r'
# and methods, e.g. fOut.write(str) writes str to fOut
# also, they are 'built in', meaning open() does not have to be imported

for line in fIn:
# 'for x in y'
# where y is any iterable object (file object, array, string, etc)
# and x is the loop variable

    # string.replace()
    line = line.replace('blue', 'brown')

    # file write
    fOut.write(line)

    # python control structures have no 'end' statement
    # the structure ends as soon as the indention reverts

#close the files
fIn.close()
fOut.close()
